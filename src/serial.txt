#include <Servo.h>

Servo myservo1;
int value = 90;
int incomingByte = 0;
void setup() {
  // 開啟 Serial Port，通訊速率為 9600bps (Bits Per Second)
  myservo1.attach(9);
  Serial.begin(9600);
}

void loop() {
  // 並輸出到 Serial Port 
  if (Serial.available()>0){
    incomingByte = Serial.read();
    switch (incomingByte) {
     case 'A':
       value = value - 30;
       break;
     case 'D':
       value = value + 30;
       break;
     case 'W':
       value = 90;
    }
    if (value>180){
      value = 180;
    }
    if (value<0){
      value=0;
    }
    Serial.print(value); 
    Serial.println("degree");
  }

    myservo1.write(value);
}
